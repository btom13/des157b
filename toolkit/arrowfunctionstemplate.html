A changed H1

This is a function that creates a bot icon HTML element and assigns all of its classes and data. It also makes it draggable, and creates a new bot in the original location when it first gets dragged. This makes it so that infinite bots can be dragged from one starting location. This is from my library liberty project.

Named Function

function make_bot() {
    const bot = document.createElement("i");
    bot.style.position = "absolute";
    bot.classList.add("fa-solid");
    bot.classList.add("fa-robot");
    bot.classList.add("draggable");
    bot.setAttribute("data-x", 0);
    bot.setAttribute("data-y", 0);
    bot.style.top = "10%";
    bot.style.left = "50%";
    bot.setAttribute("data-clicked", 0);

    gameArea.appendChild(bot);
    interact(bot).draggable({
        inertia: false,
        onstart: function (event) {
            if (event.target.getAttribute("data-clicked") == 0) {
                event.target.setAttribute("data-clicked", 1);
                make_bot();
            }
        },
        onmove: onMove,
    });
}
